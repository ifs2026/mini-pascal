// Exemplo de código válido.

program exemploValido;
var
    x, y: integer;
    z: real;
begin
    read(x, y);
    z := x + y * 2;
    if z > 10 then
        print('maior que 10', z)
    else
        print('menor ou igual a 10', z);
    for x := 1 to 5 do
        print(x);
end.
